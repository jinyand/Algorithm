class Solution {
    public long solution(int a, int b) {
        long answer = 0;
        
        if(a == b) return a;
        
        long max = (a > b) ? a : b;
        long min = (a < b) ? a : b;
        
        while(min <= max) {
            answer += min;
            min++;
        }

        return answer;
    }
}
